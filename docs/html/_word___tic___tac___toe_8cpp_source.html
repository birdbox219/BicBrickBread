280/**
281 * @brief Evaluates the potential score of each character for a given cell.
282 *
283 * @param r Row index of cell.
284 * @param c Column index of cell.
285 * @return Vector of pairs {score, character} sorted descending.
286 */
287std::vector<std::pair<int, char>> Word_XO_UI::evaluate(size_t r, size_t c)
288{
289 std::vector<std::pair<int, char>> ret;
290 int posType = ((r == 1 && c == 1) ? 1 : ((r == 0 || r == 2) && (c == 0 || c == 2)) ? 0 : 2);
291
292 for(char ch = 'A'; ch <= 'Z'; ++ch) {
293 std::set<char> s1, s2;
294 for(auto& word : Word_XO_Board::dict) {
295 if(word[posType] != ch) continue;
296 int temp = 0;
297 for(int k = 0; k < 3; ++k) {
298 if(k == posType) continue;
299 if(temp == 0) s1.insert(word[k]);
300 else s2.insert(word[k]);
301 ++temp;
302 }
303 }
304 ret.push_back({s1.size() + s2.size(), ch});
305 }
306 std::sort(ret.rbegin(), ret.rend());
307 return ret;
308}

380 if(std::get<0>(threat)!=-1) { isSafe=false; break; }

383 }
384 if(isSafe) safeMoves.push_back({r,c});
385 }
386 if(!safeMoves.empty()) {

388 int r = safeMoves[i].first, c = safeMoves[i].second;
389 int topCount = std::max(1,(int)(Word_XO_UI::score[r][c].size()*0.3));

392 }

398 bool blocksOpponent=true;

401 if(std::get<0>(opp)!=-1) blocksOpponent=false;

406 }
