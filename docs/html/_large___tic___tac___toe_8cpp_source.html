56/**
57 * @brief Construct the 5×5 board and generate all 3-in-a-row bit masks.
58 *
59 * The base Board is initialized with dummy size (1×1) because this game
60 * uses its own bitboard representation instead of the inherited matrix.
61 */
62Large_XO_Board::Large_XO_Board()
63 : Board(1, 1), emptyCell('.')
64{
65 // Initialize human-readable board (used only for display functions)
66 board.assign(5, vector<char>(5, emptyCell));
67
68 // Precompute 48 unique 3-in-a-row masks only once
69 if (win3Masks[0] == 0) {
70 size_t maskIdx = 0;
71
72 // Horizontal
73 for (int r = 0; r < 5; ++r)
74 for (int c = 0; c <= 2; ++c)
75 win3Masks[maskIdx++] =
76 (1u << (5*r+c)) | (1u << (5*r+c+1)) | (1u << (5*r+c+2));
77
78 // Vertical
79 for (int c = 0; c < 5; ++c)
80 for (int r = 0; r <= 2; ++r)
81 win3Masks[maskIdx++] =
82 (1u << (5*r+c)) | (1u << (5*(r+1)+c)) | (1u << (5*(r+2)+c));
83
84 // Diagonal ↘
85 for (int r = 0; r <= 2; ++r)
86 for (int c = 0; c <= 2; ++c)
87 win3Masks[maskIdx++] =
88 (1u << (5*r+c)) | (1u << (5*(r+1)+c+1)) | (1u << (5*(r+2)+c+2));
89
90 // Diagonal ↙
91 for (int r = 0; r <= 2; ++r)
92 for (int c = 2; c < 5; ++c)
93 win3Masks[maskIdx++] =
94 (1u << (5*r+c)) | (1u << (5*(r+1)+c-1)) | (1u << (5*(r+2)+c-2));
95 }
96}

118/**
119 * @brief Update the bitboard at (r, c) to contain symbol s.
120 *
121 * @note s = 0 clears the cell.
122 */
123bool Large_XO_Board::updateCell(size_t r, size_t c, char s) {
124 if (r >= 5 || c >= 5) return false;
125 size_t idx = 5 * r + c;
126
127 // Disallow writing into non-empty cell
128 if ((boardXO & (1u << idx)) && s != 0) return false;
129
130 uint32_t mask = ~(1u << idx);
131
132 // Clear cell
133 if (s == 0) {
134 boardX &= mask;
135 boardO &= mask;
136 boardXO &= mask;
137 --nMoves;
138 return true;
139 }
140
141 // Write cell
142 if (s == 'X') boardX |= (1u << idx);
143 else boardO |= (1u << idx);
144
145 boardXO |= (1u << idx);
146 ++nMoves;
147 return true;
148}

209void Large_XO_Board::encode(char ai, Matrix<double>& input) {
210 input.resize(25, 1);
211 char opp = (ai == 'X') ? 'O' : 'X';
212
213 for (size_t idx = 0; idx < 25; ++idx) {
214 char cell = getCell(idx / 5, idx % 5);
215 input(idx, 0) =
216 (cell == ai ? 1.0 :

353/**
354 * @brief Evaluate board using NN or terminal scoring if game is over.
355 */
356float Large_XO_AI::evaluate(Board<char>* b, Player<char>* player) {
357 auto* board = dynamic_cast<Large_XO_Board*>(b);
358 char ai = player->get_symbol();
359 char opp = (ai == 'X') ? 'O' : 'X';
360
361 // Final scoring when game ends
362 if (board->game_is_over(nullptr)) {
363 float aiScore = board->countWin(ai);
364 float oppScore = board->countWin(opp);
365
366 if (aiScore > oppScore) return 10000.0f + (aiScore - oppScore);
367 if (oppScore > aiScore) return -10000.0f + (aiScore - oppScore);
368 return 0.0f;
369 }
370
371 // Neural network evaluation
372 Matrix<double> input;
373 board->encode(ai, input);
374 Matrix<double> out = NN->predict(input);
375
376 // Choose best Q-value among empty cells
377 double bestQ = -1e9;
378 for (int idx = 0; idx < 25; ++idx)
379 if (board->getCell(idx / 5, idx % 5) == board->getEmptyCell())
380 bestQ = max(bestQ, out(idx, 0));
381
382 return static_cast<float>(bestQ);
383}

385// ----------------------------- Minimax Algorithm -----------------------------
386
387/**
388 * @brief Minimax search with alpha-beta pruning and NN move ordering.
389 */
390float Large_XO_AI::minimax(
391 bool aiTurn,
392 Player<char>* player,
393 float alpha,
394 float beta,
395 char blankCell,
396 int depth)
397{
398 auto* board = dynamic_cast<Large_XO_Board*>(player->get_board_ptr());
399 char ai = player->get_symbol();
400 char opp = (ai == 'X') ? 'O' : 'X';
401 char turn = aiTurn ? ai : opp;
402
403 // Terminal node
404 if (depth == 0 || board->game_is_over(nullptr))
405 return evaluate(board, player);
406
407 vector<pair<double,int>> moves;
408
409 // Move ordering using NN prediction
410 if (depth >= 2) {
411 Matrix<double> input;
412 board->encode(turn, input);
413 Matrix<double> out = NN->predict(input);
414
415 for (int i = 0; i < 25; ++i)
416 if (board->getCell(i / 5, i % 5) == blankCell)
417 moves.push_back({out(i, 0), i});
418
419 // Sort best-first for maximizing / minimizing
420 sort(moves.begin(), moves.end(),
421 aiTurn
422 ? [](pair<double,int>& a, pair<double,int>& b){ return a.first > b.first; }
423 : [](pair<double,int>& a, pair<double,int>& b){ return a.first < b.first; });
424 }
425 else {
426 // No ordering
427 for (int i = 0; i < 25; ++i)
428 if (board->getCell(i / 5, i % 5) == blankCell)
429 moves.push_back({0, i});
430 }
431
432 float best = aiTurn ? -1e9f : 1e9f;
433
434 // Explore moves
435 for (auto& mv : moves) {
436 int r = mv.second / 5;
437 int c = mv.second % 5;
438
439 board->updateCell(r, c, turn);
440 float score = minimax(!aiTurn, player, alpha, beta, blankCell, depth - 1);
441 board->updateCell(r, c, 0);
442
443 if (aiTurn) {
444 best = max(best, score);
445 alpha = max(alpha, score);
446 }
447 else {
448 best = min(best, score);
449 beta = min(beta, score);
450 }
451
452 // Prune branch
453 if (beta <= alpha) break;
454 }
455
456 return best;
457}
