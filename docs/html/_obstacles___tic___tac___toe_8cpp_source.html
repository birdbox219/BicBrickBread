13/**
14 * @brief Creates an empty 6×6 board, initializes the empty cell
15 * symbol, and generates all 4-in-a-row win masks if they
16 * haven't been generated already.
17 *
18 * Board layout uses 36 bits (0..35), mapping:
19 * index = row * 6 + col
20 *
21 * Bitboards used:
22 * - boardX bits for X moves
23 * - boardO bits for O moves
24 * - boardTraps bits for traps (blocked cells)
25 */
26Obstacles_Board::Obstacles_Board()
27 : Board(6, 6), emptyCell('.')
28{
29 board.assign(6, vector<char>(6, emptyCell));
30
31 // If win masks not initialized, build them now
32 if (win4Masks[0] == 0)
33 {
34 size_t maskIdx = 0;
35
36 /* ----------------------------
37 1. Horizontal win masks
38 ---------------------------- */
39 for (int r = 0; r < 6; ++r)
40 {
41 for (int c = 0; c <= 2; ++c)
42 {
43 int idx = r * 6 + c;
44 win4Masks[maskIdx++] =
45 (1ULL << idx) | (1ULL << (idx + 1)) |
46 (1ULL << (idx + 2)) | (1ULL << (idx + 3));
47 }
48 }
49
50 /* ----------------------------
51 2. Vertical win masks
52 ---------------------------- */
53 for (int c = 0; c < 6; ++c)
54 {
55 for (int r = 0; r <= 2; ++r)
56 {
57 int idx = r * 6 + c;
58 win4Masks[maskIdx++] =
59 (1ULL << idx) | (1ULL << (idx + 6)) |
60 (1ULL << (idx + 12)) | (1ULL << (idx + 18));
61 }
62 }
63
64 /* ----------------------------
65 3. Diagonal masks
66 ---------------------------- */
67 for (int r = 0; r <= 2; ++r)
68 {
69 for (int c = 0; c <= 2; ++c)
70 {
71 int idx = r * 6 + c;
72 win4Masks[maskIdx++] =
73 (1ULL << idx) | (1ULL << (idx + 7)) |
74 (1ULL << (idx + 14)) | (1ULL << (idx + 21));
75 }
76 }
77
78 /* ----------------------------
79 4. Diagonal masks
80 ---------------------------- */
81 for (int r = 0; r <= 2; ++r)
82 {
83 for (int c = 3; c < 6; ++c)
84 {
85 int idx = r * 6 + c;
86 win4Masks[maskIdx++] =
87 (1ULL << idx) | (1ULL << (idx + 5)) |
88 (1ULL << (idx + 10)) | (1ULL << (idx + 15));
89 }
90 }
91 }
92}

107char Obstacles_Board::getCell(size_t r, size_t c)
108{
109 size_t idx = r * 6 + c;
110

131/**
132 * @brief Computes a list of all free cells (not X, O, or trap).
133 *
134 * @return vector of available index positions.
135 */
136std::vector<size_t> Obstacles_Board::getAvailableMove()
137{
138 std::vector<size_t> avail;
139 avail.reserve(36);
140
141 uint64_t full = boardX | boardO | boardTraps;
142
143 for (size_t idx = 0; idx < 36; ++idx)
144 {
145 if ((full & (1ULL << idx)) == 0)
146 avail.push_back(idx);
147 }
148
149 return avail;

162/* ============================================================
163 updateCell()
164 ============================================================ */
165/**
166 * @brief Modifies the board at (r,c) with symbol s.
167 *
168 * @param r row
169 * @param c column
170 * @param s symbol ('X', 'O') or 0 to clear.
171 *
172 * Special behavior:
173 * - After a successful move, two new random traps are added.
174 */
175bool Obstacles_Board::updateCell(size_t r, size_t c, char s)
176{
177 size_t idx = r * 6 + c;
178
179 if (idx >= 36) return false;
180
181 uint64_t occupied = boardX | boardO | boardTraps;
182
183 // If placing a move and it's already occupied fail.
184 if (s != 0 && (occupied & (1ULL << idx)))
185 return false;
186
187 // Clearing a cell
188 if (s == 0)
189 {
190 uint64_t mask = ~(1ULL << idx);
191 boardX &= mask;
192 boardO &= mask;
193 boardTraps &= mask;
194
195 --nMoves;
196 return true;
197 }
198
199 // Placing X or O
200 if (s == 'X') boardX |= (1ULL << idx);
201 else boardO |= (1ULL << idx);
202
203 ++nMoves;
204
205 // Add 2 new random traps
206 auto avail = getAvailableMove();
207 if (avail.size() >= 2)
208 {
209 size_t i1 = avail[rand() % avail.size()];
210 boardTraps |= (1ULL << i1);
211
212 avail.erase(std::find(avail.begin(), avail.end(), i1));
213
214 size_t i2 = avail[rand() % avail.size()];
215 boardTraps |= (1ULL << i2);
216 }
217
218 return true;
219}
