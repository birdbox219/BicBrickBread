31 _4by4XO_Move* m = static_cast<_4by4XO_Move*>(move);
32

36 int ty = m->get_y();

38
39 // Bounds check
40 if (fx < 0 || fx >= 4 || fy < 0 || fy >= 4 || tx < 0 || tx >= 4 || ty < 0 || ty >= 4)
41 return false;
42

44 if (board[fx][fy] != s)

46

48 if (board[tx][ty] != 0)

138Move<char>* _4by4XO_AI::bestMove(Player<char>* player, char blankCell, int depth)

148 // Scan all tokens belonging to the AI
149 for (int x = 0; x < 4; x++) {
150 for (int y = 0; y < 4; y++) {
151
152 if (b->get_cell(x, y) == s) {
153
154 for (int k = 0; k < 4; k++) {
155 int nx = x + dx[k];
156 int ny = y + dy[k];
157
158 if (nx < 0 || nx >= 4 || ny < 0 || ny >= 4)
159 continue;
160
161 if (b->get_cell(nx, ny) == blankCell || b->get_cell(nx, ny) == 0) {
162 possibleMoves.push_back(new _4by4XO_Move(x, y, nx, ny, s));
163 }
164 }
165 }
166 }
167 }
168
169 if (possibleMoves.empty()) {
170 return nullptr;
171 }

174 _4by4XO_Move* chosen = possibleMoves[idx];

176 for (int i = 0; i < (int)possibleMoves.size(); i++) {

179 }
