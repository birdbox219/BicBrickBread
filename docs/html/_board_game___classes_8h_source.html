37/**
38 * @brief Base template for any board used in board games.
39 *
40 * @tparam T Type of the elements stored on the board (e.g., char, int, string).
41 *
42 * Provides core data (rows, columns, matrix) and virtual methods to be
43 * implemented by specific games like Tic-Tac-Toe, Connect4, etc.
44 */
45template <typename T>
46class Board {
47protected:
48 int rows; ///< Number of rows
49 int columns; ///< Number of columns
50 vector<vector<T>> board; ///< 2D vector for the board
51 int n_moves = 0; ///< Number of moves made
52
53public:
54 /**
55 * @brief Construct a board with given dimensions.
56 */
57 Board(int rows, int columns)
58 : rows(rows), columns(columns), board(rows, vector<T>(columns)) {}
59
60 /**
61 * @brief Virtual destructor. Frees allocated board memory.
62 */
63 virtual ~Board() {}
64
65 /**
66 * @brief Update the board with a new move.
67 * @param move The move object containing position and symbol.
68 * @return true if the move is valid and applied, false otherwise.
69 */
70 virtual bool update_board(Move<T>* move) = 0;
71
72 /** @brief Check if a player has won. */
73 virtual bool is_win(Player<T>*) = 0;
74
75 /** @brief Check if a player has lost. */
76 virtual bool is_lose(Player<T>*) = 0;
77
78 /** @brief Check if the game ended in a draw. */
79 virtual bool is_draw(Player<T>*) = 0;
80
81 /** @brief Check if the game is over. */
82 virtual bool game_is_over(Player<T>*) = 0;
83
84 /**
85 * @brief Return a copy of the current board as a 2D vector.
86 */
87 vector<vector<T>> get_board_matrix() const {
88 return board;
89 }
90
91 /** @brief Get number of rows. */
92 int get_rows() const { return rows; }
93
94 /** @brief Get number of columns. */
95 int get_columns() const { return columns; }
96
97 /** @brief Return content of cell x, y in current board. */
98
99 T get_cell(int x, int y) {
100 return board[x][y];
101 }
102};

262/**
263 * @brief Controls the flow of a board game between two players.
264 *
265 * @tparam T Type of symbol used on the board.
266 */
267template <typename T>
268class GameManager {
269 Board<T>* boardPtr; ///< Game board
270 Player<T>* players[2]; ///< Two players
271 UI<T>* ui; ///< User interface
272
273public:
274 /**
275 * @brief Construct a game manager with board, players, and UI.
276 */
277 GameManager(Board<T>* b, Player<T>* p[2], UI<T>* u)
278 : boardPtr(b), ui(u) {
279 players[0] = p[0];
280 players[1] = p[1];
281 players[0]->set_board_ptr(b);
282 players[1]->set_board_ptr(b);
283 }
284
285 /**
286 * @brief Run the main game loop until someone wins or the game ends.
287 */
288 void run() {
289 ui->display_board_matrix(boardPtr->get_board_matrix());
290 Player<T>* currentPlayer = players[0];
291
292 while (true) {
293 for (int i : {0, 1}) {
294 currentPlayer = players[i];
295 Move<T>* move = ui->get_move(currentPlayer);
296
297 while (!boardPtr->update_board(move))
298 move = ui->get_move(currentPlayer);
299
300 ui->display_board_matrix(boardPtr->get_board_matrix());
301
302 if (boardPtr->is_win(currentPlayer)) {
303 ui->display_message(currentPlayer->get_name() + " wins!");
304 return;
305 }
306 if (boardPtr->is_lose(currentPlayer)) {
307 ui->display_message(players[1 - i]->get_name() + " wins!");
308 return;
309 }
310 if (boardPtr->is_draw(currentPlayer)) {
311 ui->display_message("Draw!");
312 return;
313 }
314 }
315 }
316 }
317};
