54bool FOUR_Board::is_win(Player<char> *player)
55{
56 const char sym = player->get_symbol();
57
58 // Helper to check four in a row
59 auto all_equal = [&](char a, char b, char c, char d)
60 {
61 return a == b && b == c && c == d && a != blank_symbol;
62 };
63
64 // Horizontal checks
65 for (int i = 0; i < rows; ++i)
66 for (int j = 0; j + 3 < columns; ++j)
67 if (all_equal(board[i][j], board[i][j+1], board[i][j+2], board[i][j+3]))
68 return true;
69
70 // Vertical checks
71 for (int i = 0; i + 3 < rows; ++i)
72 for (int j = 0; j < columns; ++j)
73 if (all_equal(board[i][j], board[i+1][j], board[i+2][j], board[i+3][j]))
74 return true;
75
76 // Diagonals: up-right and down-right
77 for (int i = 0; i < rows; ++i)
78 {
79 for (int j = 0; j < columns; ++j)
80 {
81 if (i - 3 >= 0 && j + 3 < columns &&
82 all_equal(board[i][j], board[i-1][j+1], board[i-2][j+2], board[i-3][j+3])
83 && board[i][j] == sym)
84 return true;
85
86 if (i + 3 < rows && j + 3 < columns &&
87 all_equal(board[i][j], board[i+1][j+1], board[i+2][j+2], board[i+3][j+3])
88 && board[i][j] == sym)
89 return true;
90 }
91 }

188float Four_AI::minimax(bool aiTurn, Player<char> *player,
189 float alpha, float beta, char blankCell, int depth)
190{
191 auto* board = player->get_board_ptr();
192 float score = evaluate(board, player);
193
194 // Terminal conditions
195 if (score == 10) return score - depth;
196 if (score == -10) return score + depth;
197 if (!isMovesLeft(board) || depth > 6) return score;
198
199 float best = aiTurn ? INT_MIN : INT_MAX;
200 char symbol = aiTurn ? player->get_symbol()
201 : (player->get_symbol() == 'X' ? 'O' : 'X');
202
203 // Try each column
204 for (int j = 0; j < 7; j++)
205 {
206 int i = get_next_available_row(board, j);
207 if (i != -1)
208 {
209 Move<char> move(i, j, symbol);
210 board->update_board(&move);
211
212 float val = minimax(!aiTurn, player, alpha, beta, blankCell, depth+1);
213 best = aiTurn ? max(best, val) : min(best, val);
214
215 Move<char> undo(i, j, 0);
216 board->update_board(&undo);
217
218 if (aiTurn) alpha = max(alpha, best);
219 else beta = min(beta, best);
220
221 if (beta <= alpha) break;
222 }
223 }
224
225 return best;
226}
