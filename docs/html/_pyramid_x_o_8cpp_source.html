176Move<char>* PyramidXO_AI::bestMove(Player<char>* player, char blankCell, int depth) {
177 vector<pair<int,int>> available;
178 auto* board = player->get_board_ptr();
179
180 // Collect all legal pyramid cells
181 for (int r = 0; r < 5; ++r) {
182 for (int c = 0; c < 5; ++c) {
183 bool valid_shape = false;
184 if (r == 0 && c == 2) valid_shape = true;
185 else if (r == 1 && (c >= 1 && c <= 3)) valid_shape = true;
186 else if (r == 2 && (c >= 0 && c <= 4)) valid_shape = true;
187
188 if (!valid_shape) continue;
189
190 if (board->get_cell(r,c) == 0)
191 available.push_back({r,c});
192 }
193 }
194
195 // Pick random move
196 int idx = rand() % available.size();
197 return new Move<char>(available[idx].first,

199 player->get_symbol());
