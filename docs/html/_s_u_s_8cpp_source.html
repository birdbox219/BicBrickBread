135float SUS_AI::evaluate(Board<char>* board, Player<char>* player) {

138

143 score += (owner == AI) ? 10 : -10;

145 }
146
147 // Columns

151 score += (owner == AI) ? 10 : -10;
152 }
153 }
154
155 // Diagonals

158 score += (owner == AI) ? 10 : -10;

175/**
176 * @brief Minimax algorithm with alpha-beta pruning
177 */
178float SUS_AI::minimax(bool aiTurn, Player<char>* player, float alpha, float beta, char blankCell, int depth) {
179 auto* board = player->get_board_ptr();
180 float score = evaluate(board, player);
181
182 if (score >= 10) return score - depth;
183 if (score <= -10) return score + depth;
184 if (!isMovesLeft(board)) return 0;
185
186 float best = aiTurn ? INT_MIN : INT_MAX;
187 char symbol = aiTurn ? player->get_symbol() : (player->get_symbol() == 'X' ? 'O' : 'X');
188
189 for (int i = 0; i < 3; i++) {
190 for (int j = 0; j < 3; j++) {
191 if (board->get_cell(i,j)=='.') {
192 Move<char> move(i,j,symbol);
193 board->update_board(&move);
194 float val = minimax(!aiTurn, player, alpha, beta, blankCell, depth+1);
195 best = aiTurn ? max(best,val) : min(best,val);
196 Move<char> undo(i,j,0);
197 board->update_board(&undo);
198
199 if (aiTurn) alpha = max(alpha,best);
200 else beta = min(beta,best);
201 if (beta <= alpha) break;
202 }
203 }
204 if (beta <= alpha) break;
205 }
206
207 return best;
208}

213Move<char>* SUS_AI::bestMove(Player<char>* player, char blankCell, int depth) {
214 auto* board = player->get_board_ptr();
215 char AI = player->get_symbol();
216 float bestVal = INT_MIN;
217 int bestRow = -1, bestCol = -1;
218 float alpha = INT_MIN, beta = INT_MAX;
219
220 for (int i = 0; i < 3; i++) {
221 for (int j = 0; j < 3; j++) {
222 if (board->get_cell(i,j)=='.') {
223 Move<char> move(i,j,AI);
224 board->update_board(&move);
225 float moveVal = minimax(false, player, alpha, beta, blankCell, 1);
226 Move<char> undo(i,j,0);
227 board->update_board(&undo);
228
229 if (moveVal > bestVal) {
230 bestVal = moveVal;
231 bestRow = i;
232 bestCol = j;
233 }
234 alpha = max(alpha,bestVal);
235 }
236 }
237 }
238
239 return new Move<char>(bestRow,bestCol,AI);
