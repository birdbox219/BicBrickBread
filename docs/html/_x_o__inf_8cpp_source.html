26/**
27 * @brief Apply or undo a move on the board.
28 *
29 * @param move Pointer to the move to apply. If `move->get_symbol() == 0`, it
30 * represents an undo operation.
31 * @return True if the move was successfully applied, false if invalid.
32 *
33 * @details
34 * - For normal moves (`mark != 0`):
35 * - Validates cell coordinates and occupancy.
36 * - Updates the board and history.
37 * - Increments move counter `cnt` modulo 3.
38 * - Every third move, oldest move is pushed to `undoHistory`.
39 * - For undo moves (`mark == 0`):
40 * - Decrements move counter and resets cell.
41 * - Restores move from `undoHistory` if needed.
42 * - Updates `cnt` modulo 3.
43 */
44bool XO_inf_Board::update_board(Move<char>* move) {
45 int x = move->get_x();
46 int y = move->get_y();
47 char mark = move->get_symbol();
48
49 if (x < 0 || x >= rows || y < 0 || y >= columns || (board[x][y] != blank_symbol && mark != 0))
50 return false;
51
52
53
54 if (mark == 0) { // Undo move
55 --n_moves;
56 board[x][y] = blank_symbol;
57 history.pop_back();
58 if(cnt == 0 && !undoHistory.empty() ) {
59 history.push_front(undoHistory.top());
60 board[undoHistory.top().x][undoHistory.top().y] = undoHistory.top().symbol;
61 undoHistory.pop();
62 }
63 cnt = (cnt - 1 + 3) % 3;
64 return true;
65 }
66
67 // Apply normal move
68 board[x][y] = mark;
69 ++n_moves;
70 history.push_back({x, y, mark});
71 ++cnt;
72 cnt %= 3;
73
74 if (cnt == 0 && !history.empty()) {
75 undoHistory.push(history.front());
76 board[history.front().x][history.front().y] = blank_symbol;
77 history.pop_front();
78 }
79
80 return true;
81}

150/**
151 * @brief Minimax algorithm with alpha-beta pruning.
152 *
153 * @param aiTurn True if it's AI's turn.
154 * @param player Pointer to the AI player.
155 * @param alpha Alpha value for pruning.
156 * @param beta Beta value for pruning.
157 * @param blankCell Symbol representing an empty cell.
158 * @param depth Maximum recursion depth.
159 * @return Best minimax score from the current board position.
160 */
161float XO_inf_AI::minimax(bool aiTurn, Player<char>* player, float alpha, float beta, char blankCell, int depth)
162{
163 auto* board = dynamic_cast<XO_inf_Board*>(player->get_board_ptr());
164 if (!board || !player) throw invalid_argument("Invalid board or player in minimax()");
165
166 float score = evaluate(board, player);
167 if (score == 10 || score == -10 || board->is_draw(player) || depth == 0)
168 return score;
169
170 char ai = player->get_symbol();
171 char opp = (ai == 'X') ? 'O' : 'X';
172 char turn = aiTurn ? ai : opp;
173 float best = aiTurn ? -INF : INF;
174
175 for (int i = 0; i < 3; ++i)
176 for (int j = 0; j < 3; ++j)
177 if (board->get_cell(i, j) == blankCell)
178 {
179 Move<char> move(i, j, turn);
180 board->update_board(&move);
181
182 float val = minimax(!aiTurn, player, alpha, beta, blankCell, depth - 1);
183
184 move = Move<char>(i, j, 0); // Undo move
185 board->update_board(&move);
186
187 if (aiTurn) { best = max(best, val); alpha = max(alpha, val); }
188 else { best = min(best, val); beta = min(beta, val); }
189
190 if (beta <= alpha) return best;
191 }
192
193 return best;
194}

198 *

200 * @param blankCell Symbol representing empty cells.

206Move<char>* XO_inf_AI::bestMove(Player<char>* player, char blankCell, int depth)
207{

209 if (!board || !player) throw invalid_argument("Invalid board or player in bestMove()");
210
211 char ai = player->get_symbol();
212 float bestVal = -INF;
213 int bestR = -1, bestC = -1;
214
215 for (int i = 0; i < 3; ++i)
216 for (int j = 0; j < 3; ++j)
217 if (board->get_cell(i, j) == blankCell)
218 {
219 Move<char> move(i, j, ai);
220 board->update_board(&move);
221
222 float val = minimax(false, player, -INF, INF, blankCell, depth - 1);
223
224 move = Move<char>(i, j, 0); // Undo move
225 board->update_board(&move);
226
227 if (val > bestVal) { bestVal = val; bestR = i; bestC = j; }
228 }
229
230 if (bestR == -1 || bestC == -1)
