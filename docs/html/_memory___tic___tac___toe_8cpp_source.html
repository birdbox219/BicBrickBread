31// Update board with a player's move
32bool Memory_Board::update_board(Move<char>* move)
33{
34 if (!move) throw invalid_argument("Null move pointer");
35
36 int x = move->get_x();
37 int y = move->get_y();
38 char mark = move->get_symbol();
39
40 // Validate move coordinates
41 if (x < 0 || x >= rows || y < 0 || y >= columns)
42 throw out_of_range("Move coordinates out of bounds");
43
44 // Undo move if mark is blank or zero
45 if (mark == 0 || mark == blank_symbol)
46 {
47 if (board[x][y] != blank_symbol) n_moves--; // decrement move count
48 board[x][y] = blank_symbol;
49 return true;
50 }
51
52 // Apply move if cell is empty
53 if (board[x][y] == blank_symbol)
54 {
55 board[x][y] = toupper(mark);
56 n_moves++;
57 return true;
58 }
59
60 return false; // Cell already occupied
61}

63// Check if the given player has a winning line
64bool Memory_Board::is_win(Player<char>* player)
65{
66 if (!player) throw invalid_argument("Null player pointer");
67 char sym = player->get_symbol();
68
69 // Lambda to check if three cells are equal and non-blank
70 auto all_equal = [&](char a, char b, char c) { return a == b && b == c && a != blank_symbol; };
71
72 // Check all rows and columns
73 for (int i = 0; i < rows; ++i)
74 if ((all_equal(board[i][0], board[i][1], board[i][2]) && board[i][0] == sym) ||
75 (all_equal(board[0][i], board[1][i], board[2][i]) && board[0][i] == sym))
76 return true;
77
78 // Check both diagonals
79 if ((all_equal(board[0][0], board[1][1], board[2][2]) && board[1][1] == sym) ||
80 (all_equal(board[0][2], board[1][1], board[2][0]) && board[1][1] == sym))
81 return true;
82
83 return false;
84}

136// Minimax algorithm with alpha-beta pruning
137float Memory_AI::minimax(bool aiTurn, Player<char>* player, float alpha, float beta, char blankCell, int depth)
138{
139 auto* board = dynamic_cast<Memory_Board*>(player->get_board_ptr());
140 if (!board || !player) throw invalid_argument("Invalid board or player in minimax()");
141
142 float score = evaluate(board, player);
143 if (score == 10 || score == -10 || board->is_draw(player) || depth == 0)
144 return score;
145
146 char ai = player->get_symbol();
147 char opp = (ai == 'X') ? 'O' : 'X';
148 char turn = aiTurn ? ai : opp;
149
150 float best = aiTurn ? -INF : INF;
151
152 // Iterate through all cells
153 for (int i = 0; i < 3; ++i)
154 for (int j = 0; j < 3; ++j)
155 if (board->get_cell(i, j) == blankCell)
156 {
157 // Try move
158 Move<char> move(i, j, turn);
159 board->update_board(&move);
160
161 // Recursive evaluation
162 float val = minimax(!aiTurn, player, alpha, beta, blankCell, depth - 1);
163
164 // Undo move
165 move = Move<char>(i, j, 0);
166 board->update_board(&move);
167
168 // Update best score and alpha/beta
169 if (aiTurn) { best = max(best, val); alpha = max(alpha, val); }
170 else { best = min(best, val); beta = min(beta, val); }
171
172 // Alpha-beta pruning
173 if (beta <= alpha) return best;
174 }
175
176 return best;
177}

179// Compute the best move for AI
180Move<char>* Memory_AI::bestMove(Player<char>* player, char blankCell, int depth)
181{
182 auto* board = dynamic_cast<Memory_Board*>(player->get_board_ptr());
183 if (!board || !player) throw invalid_argument("Invalid board or player in bestMove()");
184
185 char ai = player->get_symbol();
186 float bestVal = -INF;
187 int bestR = -1, bestC = -1;
188
189 // Evaluate all possible moves
190 for (int i = 0; i < 3; ++i)
191 for (int j = 0; j < 3; ++j)
192 if (board->get_cell(i, j) == blankCell)
193 {
194 Move<char> move(i, j, ai);
195 board->update_board(&move);
196
197 float val = minimax(false, player, -INF, INF, blankCell, depth - 1);
198
199 // Undo move
200 move = Move<char>(i, j, 0);
201 board->update_board(&move);
202
203 if (val > bestVal) { bestVal = val; bestR = i; bestC = j; }
204 }
205
206 if (bestR == -1 || bestC == -1)
207 throw runtime_error("No valid moves left for AI");
208
209 return new Move<char>(bestR, bestC, ai);
210}
